#!/bin/bash

echo "=========================================="
echo "Лабораторная работа №3: Циклы в Bash"
echo "=========================================="
echo ""

echo "1. Цикл for - перебор чисел от 1 до 5:"
echo "------------------------------------------"

for i in 1 2 3 4 5
do
    echo "   Итерация номер: $i"
done

echo ""

echo "2. Цикл for с диапазоном {1..5}:"
echo "------------------------------------------"

for i in {1..5}
do
    echo "   Квадрат числа $i = $((i * i))"
done

echo ""

echo "3. Цикл while - счётчик от 1 до 5:"
echo "------------------------------------------"

counter=1
while [ $counter -le 5 ]
do
    echo "   Счётчик: $counter"
    ((counter++))
done

echo ""

echo "4. Цикл while - перебор строк из переменной:"
echo "------------------------------------------"

fruits="яблоко груша апельсин банан"
echo "Фрукты: $fruits"
echo ""

index=1
for fruit in $fruits
do
    echo "   Фрукт $index: $fruit"
    ((index++))
done

echo ""

echo "5. Цикл until - выполняется до достижения 5:"
echo "------------------------------------------"

num=1
until [ $num -gt 5 ]
do
    echo "   Число: $num (until остановится когда num > 5)"
    ((num++))
done

echo ""

echo "6. Вложенные циклы (таблица умножения на 2 и 3):"
echo "------------------------------------------"

for i in 2 3
do
    echo "   Таблица умножения на $i:"
    for j in 1 2 3 4 5
    do
        result=$((i * j))
        echo "      $i x $j = $result"
    done
    echo ""
done

echo "7. Создание файлов через цикл:"
echo "------------------------------------------"

for i in {1..3}
do
    filename="test_file_$i.txt"
    echo "Содержимое файла $i" > $filename
    echo "   Создан файл: $filename"
done

echo ""
echo "Содержимое созданных файлов:"
ls -la test_file_*.txt

echo ""
echo "=========================================="
echo "Лабораторная работа №3 завершена"
echo "=========================================="
